% Define a custom command for the hexagonal icon
\newcommand{\hexicon}[2]{%
  \smash{% Tells LaTeX to ignore the vertical height of the icon
    \begin{tikzpicture}[baseline={([yshift=-0.6ex]char.center)}]
    \node[
      draw=#1,
      fill=white,
      shape=regular polygon,
      regular polygon sides=6,
      inner sep=-1.0pt,
      minimum size=0.9cm,
      line width=1.5pt
    ] (char) {\color{#1}\large#2};
    \end{tikzpicture}%
  }%
}

\NewDocumentCommand{\cvsection}{ m m m }{%
  \vspace{3ex}
  \noindent
  \begin{minipage}[c]{1.0cm}
    \centering \hexicon{#1}{#2} % Passes the color and icon to your updated hexicon
  \end{minipage}%
  \begin{minipage}[c]{\dimexpr\linewidth-1.0cm\relax}
    {\Large\bfseries\color{cvDarkgray}\hspace{1.1ex}#3}\par
    \vspace{-1.6ex}
    \color{#1!70}\rule{\linewidth}{1.5pt}% Uses the passed color at 70% opacity for the line
  \end{minipage}%
  \par\vspace{1.5ex}
}
